  var DAY_MAP={'Su':0,'M':1,'T':2,'W':3,'Th':4,'F':5,'Sa':6};
  function getClassesForDate(y,m,d){
    var dow=new Date(y,m,d).getDay();
    var dateStr=dKey(y,m,d);
    var all=[];
    ['fall','winter','spring'].forEach(function(q){
      var range=getQuarterRange(q);
      if(range.start&&range.end){if(dateStr<range.start||dateStr>range.end)return;}
      else if(range.start&&!range.end){if(dateStr<range.start)return;}
      else if(!range.start&&range.end){if(dateStr>range.end)return;}
      (schedule[q]||[]).forEach(function(c){
        var mainMatch=(c.days||[]).some(function(day){return DAY_MAP[day]===dow;});
        if(mainMatch)all.push({cls:c,quarter:q,isDisc:false});
        if(c.discussion&&c.discussion.enabled&&(c.discussion.days||[]).some(function(day){return DAY_MAP[day]===dow;})){
          all.push({cls:c,quarter:q,isDisc:true,discLabel:c.discussion.type||'Discussion'});
        }
      });
    });
    return all;
  }

  function parseTimeToHHMM(t){
    if(!t)return{h:'',m:'',ampm:'AM'};
    var p=t.split(':');
    var h24=+p[0]||0,m=p[1]||'00';
    var ampm=h24>=12?'PM':'AM';
    var h12=h24%12||12;
    return{h:String(h12),m:m,ampm:ampm};
  }
  function buildTime(h,m,ampm){
    if(h===''&&m==='')return '';
    var h12=+h||12;
    var h24=ampm==='PM'?(h12===12?12:h12+12):(h12===12?0:h12);
    return String(h24).padStart(2,'0')+':'+String(+m||0).padStart(2,'0');
  }
  function formatDisplayTime(t){
    if(!t)return '';
    var parsed=parseTimeToHHMM(t);
    return parsed.h+':'+(parsed.m.padStart(2,'0'))+' '+parsed.ampm;
  }

  function fireNotif(title, body){
    if(Notification.permission!=='granted') return;
    // try service worker first (works in background)
    if('serviceWorker' in navigator && navigator.serviceWorker.controller){
      navigator.serviceWorker.controller.postMessage({type:'NOTIFY', title:title, body:body});
    } else {
      try{ new Notification(title, {body:body}); }catch(e){}
    }
    showToast(title, body);
  }

// 1. Load saved data when the page opens
document.addEventListener("DOMContentLoaded", () => {
const inputs = document.querySelectorAll("input, textarea");
inputs.forEach(input => {
if (input.id) {
const savedValue = localStorage.getItem(input.id);
if (savedValue) {
input.value = savedValue;
}
}
});
});

if (event.target.id) {
localStorage.setItem(event.target.id, event.target.value);
}
});
